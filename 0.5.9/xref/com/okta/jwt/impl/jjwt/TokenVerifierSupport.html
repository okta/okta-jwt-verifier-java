16  package com.okta.jwt.impl.jjwt;
17  
18  import com.okta.jwt.Jwt;
19  import com.okta.jwt.JwtVerificationException;
20  import com.okta.jwt.impl.DefaultJwt;
21  import io.jsonwebtoken.Claims;
22  import io.jsonwebtoken.Jws;
23  import io.jsonwebtoken.JwtException;
24  import io.jsonwebtoken.JwtHandlerAdapter;
25  import io.jsonwebtoken.JwtParser;
26  import io.jsonwebtoken.Jwts;
27  import io.jsonwebtoken.SignatureAlgorithm;
28  import io.jsonwebtoken.SigningKeyResolver;
29  import io.jsonwebtoken.UnsupportedJwtException;
30  
31  import java.time.Clock;
32  import java.time.Duration;
33  import java.util.Date;
34  
35  abstract class TokenVerifierSupport {
36  
37      private final SigningKeyResolver keyResolver;
38      private final String issuer;
39      private final Duration leeway;
40      private final JwtParser jwtParser;
41      private final Clock clock;
42  
43      TokenVerifierSupport(String issuer,
44                           Duration leeway,
45                           SigningKeyResolver signingKeyResolver) {
46          this.issuer = issuer;
47          this.leeway = leeway;
48          this.keyResolver = new IssuerMatchingSigningKeyResolver(issuer, signingKeyResolver);
49          this.clock = Clock.systemDefaultZone();
50          this.jwtParser = buildJwtParser();
51      }
52  
53      TokenVerifierSupport(String issuer,
54                           Duration leeway,
55                           SigningKeyResolver signingKeyResolver,
56                           Clock clock) {
57          this.issuer = issuer;
58          this.leeway = leeway;
59          this.keyResolver = new IssuerMatchingSigningKeyResolver(issuer, signingKeyResolver);
60          this.clock = clock;
61          this.jwtParser = buildJwtParser();
62      }
63  
64      protected JwtParser buildJwtParser() {
65          io.jsonwebtoken.Clock jwtClock = () -> Date.from(clock.instant());
66          return Jwts.parser()
67                  .setSigningKeyResolver(keyResolver)
68                  .requireIssuer(issuer)
69                  .setAllowedClockSkewSeconds(leeway.getSeconds())
70                  .setClock(jwtClock)
71                  .build();
72      }
73  
74      protected Jwt decode(String token, JwtParser parser, ClaimsValidator claimsValidator) throws JwtVerificationException {
75  
76          if (!parser.isSigned(token)) {
77              throw new JwtVerificationException("Token did not contain signature");
78          }
79  
80          try {
81              Jws<Claims> jwt = parser.parse(token, new OktaJwtHandler(claimsValidator));
82              return new DefaultJwt(token,
83                      jwt.getBody().getIssuedAt().toInstant(),
84                      jwt.getBody().getExpiration().toInstant(),
85                      jwt.getBody());
86          } catch (JwtException e) {
87              throw new JwtVerificationException("Failed to parse token", e);
88          }
89      }
90  
91      SigningKeyResolver getKeyResolver() {
92          return keyResolver;
93      }
94  
95      String getIssuer() {
96          return issuer;
97      }
98  
99      Duration getLeeway() {
100         return leeway;
101     }
102 
103     JwtParser parser() {
104         return jwtParser;
105     }
106 
107     Clock getClock() {
108         return clock;
109     }
110 
111     static class OktaJwtHandler extends JwtHandlerAdapter<Jws<Claims>> {
112 
113         private final ClaimsValidator claimsValidator;
114 
115         protected OktaJwtHandler(ClaimsValidator claimsValidator) {
116             this.claimsValidator = claimsValidator;
117         }
118 
119         @Override
120        public Jws<Claims> onClaimsJws(Jws<Claims> jws) {
121 
122            // validate alg
123            String alg = jws.getHeader().getAlgorithm();
124            if(!SignatureAlgorithm.RS256.getValue().equals(alg)) {
125                throw new UnsupportedJwtException("JWT Header 'alg' of [" + alg + "] is not supported, only RSA25 signatures are supported");
126            }
127 
128            claimsValidator.validateClaims(jws);
129            return jws;
130         }
131     }
132 }
