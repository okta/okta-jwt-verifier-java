16  package com.okta.jwt.impl.jjwt;
17  
18  import com.okta.jwt.AccessTokenVerifier;
19  import com.okta.commons.lang.Assert;
20  import org.slf4j.Logger;
21  import org.slf4j.LoggerFactory;
22  
23  /**
24   * Classes in this `impl` implementation package may change in NON backward compatible way, and should ONLY be used as
25   * a "runtime" dependency.
26   */
27  public final class JjwtAccessTokenVerifierBuilder extends BaseVerifierBuilderSupport<AccessTokenVerifier.Builder, AccessTokenVerifier>
28                                                    implements AccessTokenVerifier.Builder {
29  
30      private final static Logger log = LoggerFactory.getLogger(JjwtAccessTokenVerifierBuilder.class);
31      private String audience = "api://default";
32  
33      public AccessTokenVerifier.Builder setAudience(String audience) {
34          Assert.notNull(audience, "audience cannot be null");
35          this.audience = audience;
36          return this;
37      }
38  
39      @Override
40      protected void validate() {
41          super.validate();
42          if (audience == null || audience.isEmpty()) {
43              throw new IllegalArgumentException("audience cannot be null or empty");
44          }
45  
46          if (!getIssuer().matches(".*/oauth2/.*")) {
47              log.warn("Decoding access tokens from this issuer '{}' may not be possible. Your issuer URL should be in the format of 'https://{yourOktaDomain}/oauth2/qualifier'", getIssuer());
48          }
49  
50          if (getConnectionTimeout().isNegative()) {
51              throw new IllegalArgumentException("timeout cannot be < 0");
52          }
53      }
54  
55      @Override
56      public AccessTokenVerifier build() {
57          validate();
58          return new JjwtAccessTokenVerifier(getIssuer(), audience, getLeeway(), signingKeyResolver(), getClock());
59      }
60  }
