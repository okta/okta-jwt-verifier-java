16  package com.okta.jwt.impl.jjwt;
17  
18  import com.okta.commons.configcheck.ConfigurationValidator;
19  import com.okta.commons.http.authc.DisabledAuthenticator;
20  import com.okta.commons.http.config.HttpClientConfiguration;
21  import com.okta.jwt.VerifierBuilderSupport;
22  import com.okta.jwt.impl.http.HttpClient;
23  import com.okta.jwt.impl.http.OktaCommonsHttpClient;
24  import io.jsonwebtoken.SigningKeyResolver;
25  
26  import java.net.MalformedURLException;
27  import java.net.URL;
28  import java.time.Clock;
29  import java.time.Duration;
30  import java.util.Objects;
31  
32  abstract class BaseVerifierBuilderSupport<B extends VerifierBuilderSupport, R> implements VerifierBuilderSupport<B, R> {
33  
34      private String issuer;
35      private Duration leeway = Duration.ofMinutes(2);
36      private Duration connectionTimeout = Duration.ofSeconds(1);
37      private String proxyHost = null;
38      private int proxyPort;
39      private String proxyUsername = null;
40      private String proxyPassword = null;
41      private int retryMaxAttempts = 2; /* based on SDK spec */
42      private Duration retryMaxElapsed = Duration.ofSeconds(10);
43      private Clock clock = Clock.systemDefaultZone();
44      private boolean preloadSigningKeys = false;
45  
46      String getIssuer() {
47          return issuer;
48      }
49  
50      public B setIssuer(String issuer) {
51  
52          // delay validation until 'validate' method is called
53          if (issuer != null) {
54              // trim and remove any trailing slash
55              this.issuer = issuer.trim().replaceAll("/$", "");
56          }
57          return self();
58      }
59  
60      Duration getLeeway() {
61          return leeway;
62      }
63  
64      public B setLeeway(Duration leeway) {
65          if (leeway == null || leeway.toMillis() < 0) {
66              throw new IllegalArgumentException("leeway must not be null or less than zero");
67          }
68          this.leeway = leeway;
69          return self();
70      }
71  
72      Duration getConnectionTimeout() {
73          return connectionTimeout;
74      }
75  
76      public B setConnectionTimeout(Duration connectionTimeout) {
77          this.connectionTimeout = connectionTimeout;
78          return self();
79      }
80  
81      public String getProxyHost() {
82          return proxyHost;
83      }
84  
85      @Override
86      public B setProxyHost(String proxyHost) {
87          this.proxyHost = proxyHost;
88          return self();
89      }
90  
91      public int getProxyPort() {
92          return proxyPort;
93      }
94  
95      @Override
96      public B setProxyPort(int proxyPort) {
97          this.proxyPort = proxyPort;
98          return self();
99      }
100 
101     public String getProxyUsername() {
102         return proxyUsername;
103     }
104 
105     @Override
106     public B setProxyUsername(String proxyUsername) {
107         this.proxyUsername = proxyUsername;
108         return self();
109     }
110 
111     public String getProxyPassword() {
112         return proxyPassword;
113     }
114 
115     @Override
116     public B setProxyPassword(String proxyPassword) {
117         this.proxyPassword = proxyPassword;
118         return self();
119     }
120 
121     public int getRetryMaxAttempts() {
122         return retryMaxAttempts;
123     }
124 
125     public B setRetryMaxAttempts(int retryMaxAttempts) {
126         this.retryMaxAttempts = retryMaxAttempts;
127         return self();
128     }
129 
130     public Duration getMaxHttpRetryElapsed() {
131         return retryMaxElapsed;
132     }
133 
134     public B setRetryMaxElapsed(Duration retryMaxElapsed) {
135         this.retryMaxElapsed = retryMaxElapsed;
136         return self();
137     }
138 
139     public Clock getClock() {
140         return clock;
141     }
142 
143     public B setClock(Clock clock) {
144         this.clock = clock;
145         return self();
146     }
147 
148    public boolean getPreloadSigningKeys() {
149         return preloadSigningKeys;
150     }
151 
152     @Override
153     public B setPreloadSigningKeys(boolean preloadSigningKeys) {
154         this.preloadSigningKeys = preloadSigningKeys;
155         return self();
156     }
157 
158 
159     @SuppressWarnings("unchecked")
160     protected B self() {
161         return (B) this;
162     }
163 
164     protected void validate() {
165         ConfigurationValidator.assertIssuer(issuer);
166     }
167 
168     protected String resolveKeysEndpoint(String issuer) {
169         return  issuer.matches(".*/oauth2/.*")
170                     ? issuer + "/v1/keys"
171                     : issuer + "/oauth2/v1/keys";
172     }
173 
174     protected SigningKeyResolver signingKeyResolver() {
175         try {
176            RemoteJwkSigningKeyResolver remoteJwkSigningKeyResolver =  new RemoteJwkSigningKeyResolver(
177                             new URL(resolveKeysEndpoint(getIssuer())),
178                             httpClient());
179            //preload keys during start up. so that if the call the issuer keys fails, its not a runtime exception.
180             if (getPreloadSigningKeys()) {
181                 remoteJwkSigningKeyResolver.updateKeys();
182             }
183             return remoteJwkSigningKeyResolver;
184         } catch (MalformedURLException e) {
185             throw new IllegalStateException("Invalid issuer URL in configuration");
186         }
187     }
188 
189     protected HttpClient httpClient() {
190         HttpClientConfiguration httpClientConfiguration = new HttpClientConfiguration();
191         httpClientConfiguration.setRequestAuthenticator(new DisabledAuthenticator());
192         httpClientConfiguration.setConnectionTimeout((int) getConnectionTimeout().getSeconds());
193         httpClientConfiguration.setRetryMaxAttempts(getRetryMaxAttempts()); // number of retry attempts
194         httpClientConfiguration.setRetryMaxElapsed((int) getMaxHttpRetryElapsed().getSeconds()); // number of seconds
195         httpClientConfiguration.setProxyHost(getProxyHost());
196         httpClientConfiguration.setProxyPort(getProxyPort());
197         httpClientConfiguration.setProxyUsername(getProxyUsername());
198         httpClientConfiguration.setProxyPassword(getProxyPassword());
199         return new OktaCommonsHttpClient(httpClientConfiguration);
200     }
201 
202     @Override
203     public boolean equals(Object o) {
204         if (this == o) return true;
205         if (o == null || getClass() != o.getClass()) return false;
206         BaseVerifierBuilderSupport<?, ?> that = (BaseVerifierBuilderSupport<?, ?>) o;
207         return proxyPort == that.proxyPort &&
208                 retryMaxAttempts == that.retryMaxAttempts &&
209                 Objects.equals(issuer, that.issuer) &&
210                 Objects.equals(leeway, that.leeway) &&
211                 Objects.equals(connectionTimeout, that.connectionTimeout) &&
212                 Objects.equals(proxyHost, that.proxyHost) &&
213                 Objects.equals(proxyUsername, that.proxyUsername) &&
214                 Objects.equals(proxyPassword, that.proxyPassword) &&
215                 Objects.equals(retryMaxElapsed, that.retryMaxElapsed) &&
216                 Objects.equals(clock, that.clock);
217     }
218 
219     @Override
220     public int hashCode() {
221         return Objects.hash(issuer, leeway, connectionTimeout, proxyHost, proxyPort, proxyUsername, proxyPassword, retryMaxAttempts, retryMaxElapsed, clock);
222     }
223 }
